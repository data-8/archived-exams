\vskip .1in

{\bf Example Expression:} \lsi+make_array(1,2,3,4,5) == 3+

\vskip .1in

{\bf Example Answer:} \lsi+array([False, False, True, False, False])+

\vskip .5in

\solution{\lsi+array([1,6])+
